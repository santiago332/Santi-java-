import java.util.Scanner;

public class ScannerWarmup {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        
        System.out.println("Emri: ");
        String emri = sc.nextLine();  
        
        System.out.println("Mosha: ");
        int mosha = sc.nextInt(); 
        
        System.out.println("Koeficienti: ");
        double koeficient = sc.nextDouble();  
        sc.nextLine(); 
        
        System.out.println("Përshëndetje, " + emri + "! Mosha: " + mosha + ", Koeficienti: " + koeficient + ".");
       
    }
}
